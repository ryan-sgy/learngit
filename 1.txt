    #include<string>
    #include<iostream>
    using namespace std;

    void MemeryArray(int a[], int n, int b[], int m, int c[])
    {
        int i, j, k;
        i = j = k = 0;
        while (i < n && j < m)
        {
            if (a[i] < b[j])
                c[k++] = a[i++];
            else
                c[k++] = b[j++];
        }
        while (i < n)
            c[k++] = a[i++];
        while (j < m)
            c[k++] = b[j++];
    }
    int main()
    {
        int N,M;
        scanf("%d%d",&M,&N);
        int *A = new int[N];
        int *B = new int[M];
        for (int i = 0; i < N; i++)
            cin >> A[i];
        for (int i = 0; i < M; i++)
            cin >> B[i];
        int *C = new int[M + N];
        MemeryArray(A, N, B, M, C);
        for (int i = 0; i < M + N; i++)
            cout << C[i]<< " ";
        return 0;
    }
